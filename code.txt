// Get Allowed Admin IDs from dedicated sheet
function getAllowedAdminIds() {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName('AllowedAdmins');
    if (!sheet) {
      sheet = ss.insertSheet('AllowedAdmins');
      sheet.appendRow([MAIN_ADMIN_EMP_ID]);
      return [MAIN_ADMIN_EMP_ID];
    }
    var data = sheet.getDataRange().getValues();
    var ids = data.map(function(row) { return (row[0] || "").toString().trim().toUpperCase(); }).filter(function(id) { return id.length > 0; });
    
    if (ids.length === 0) return [MAIN_ADMIN_EMP_ID];
    
    // ALWAYS ensure MAIN_ADMIN_EMP_ID is in the list (protection layer)
    if (ids.indexOf(MAIN_ADMIN_EMP_ID) === -1) {
      ids.unshift(MAIN_ADMIN_EMP_ID);
      setAllowedAdminIds(ids);
    }
    
    return ids;
  } catch (e) {
    return [MAIN_ADMIN_EMP_ID];
  }
}

  if (!sheet) {
    sheet = ss.insertSheet('AllowedAdmins');
  }
  
  var normalizedIds = ids.map(function(id) { return id.toString().trim().toUpperCase(); });
  if (normalizedIds.indexOf(MAIN_ADMIN_EMP_ID) === -1) {

  }
}

// Simple hash function for passwords
function hashPassword(password) {
  var hash = 0;
  for (var i = 0; i < password.length; i++) {
    var char = password.charCodeAt(i);
    hash = ((hash << 5) - hash) + char;
    hash = hash & hash;
  }
  return hash.toString();
}

// Check if Emp ID is allowed for registration
function isEmpIdAllowed(empId, accessLevel) {
  var normalizedId = (empId || "").toString().trim().toUpperCase();
  if (accessLevel === 'admin') {
    var adminIds = getAllowedAdminIds();
    return adminIds.indexOf(normalizedId) !== -1;
  } else if (accessLevel === 'crewcontroller') {
    var ccIds = getAllowedCcIds();
    return ccIds.indexOf(normalizedId) !== -1;
  }
  return false;
}

// Register new user
function registerUser(empId, name, password, accessCode, accessLevel) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var usersSheet = setupUsersSheet();
    
    // Verify access code
    if (accessCode !== ADMIN_ACCESS_CODE) {
      return { success: false, message: "Invalid Access Code!" };
    }
    
    // Verify Emp ID is allowed
    if (!isEmpIdAllowed(empId, accessLevel)) {
      return { success: false, message: "Emp ID not authorized for this access level!" };
    }
    
    var normalizedId = (empId || "").toString().trim().toUpperCase();
    var normalizedName = (name || "").toString().trim();
    
    // Check if Emp ID already registered
    var userData = usersSheet.getDataRange().getValues();
    for (var i = 1; i < userData.length; i++) {
      if (userData[i][0].toString().toUpperCase() === normalizedId) {
        return { success: false, message: "Emp ID already registered! Please login." };
      }
    }
    
    // Hash password and save
    var passwordHash = hashPassword(password);
    var now = new Date().toLocaleString('en-IN', { timeZone: 'Asia/Kolkata' });
    
    usersSheet.appendRow([normalizedId, normalizedName, passwordHash, accessLevel, now, '']);
    
    return { success: true, message: "Registration successful! Please login with Emp ID and Password." };
  } catch (e) {
    return { success: false, message: "Error: " + e.toString() };
  }
}

// Login user

  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var usersSheet = setupUsersSheet();
    
    var normalizedId = (empId || "").toString().trim().toUpperCase();
    var passwordHash = hashPassword(password);
    
    var userData = usersSheet.getDataRange().getValues();
    for (var i = 1; i < userData.length; i++) {
      if (userData[i][0].toString().toUpperCase() === normalizedId) {
        // Check password
        if (userData[i][2].toString() === passwordHash) {
          // Update last login
          var now = new Date().toLocaleString('en-IN', { timeZone: 'Asia/Kolkata' });
          usersSheet.getRange(i + 1, 6).setValue(now);
          
          return { 
            success: true, 
            empId: userData[i][0], 
            name: userData[i][1], 
            accessLevel: (userData[i][3] || "").toString().toLowerCase(),
            message: "Login successful!" 
          };
        } else {
          return { success: false, message: "Invalid Password!" };
        }
      }
    }
    
    return { success: false, message: "Emp ID not found! Please register first." };
  } catch (e) {
    return { success: false, message: "Error: " + e.toString() };
  }
}

// Add allowed Emp ID (Only admins can do this, secret code required for admin access)
function addAllowedEmpId(empId, accessLevel, secretCode, requesterEmpId) {
  try {
    var adminIds = getAllowedAdminIds();
    var requesterNormalized = (requesterEmpId || "").toString().trim().toUpperCase();
    if (adminIds.indexOf(requesterNormalized) === -1) {
      return { success: false, message: "Only admins can modify access!" };
    }
    
    var normalizedId = (empId || "").toString().trim().toUpperCase();
    
    if (normalizedId === MAIN_ADMIN_EMP_ID) {
      return { success: false, message: "Cannot modify main admin access!" };
    }
    
    if (accessLevel === 'admin') {
      // Secret code required for admin access modification
      if ((secretCode || "").toString().trim().toLowerCase() !== ADMIN_SECRET_CODE) {
        return { success: false, message: "Invalid Secret Code!" };
      }
      if (adminIds.indexOf(normalizedId) !== -1) {
        return { success: false, message: "Emp ID already in admin list!" };
      }
      adminIds.push(normalizedId);
      setAllowedAdminIds(adminIds);
      return { success: true, message: normalizedId + " added to Admin access!" };
    } else if (accessLevel === 'crewcontroller') {
      var ccIds = getAllowedCcIds();
      if (ccIds.indexOf(normalizedId) !== -1) {
        return { success: false, message: "Emp ID already in Crew Controller list!" };
      }
      ccIds.push(normalizedId);
      setAllowedCcIds(ccIds);
      return { success: true, message: normalizedId + " added to Crew Controller access!" };
    }
    
    return { success: false, message: "Invalid access level!" };
  } catch (e) {
    return { success: false, message: "Error: " + e.toString() };
  }
}

function getVisitorStats() {
  try {
    var sheet = getOrCreateVisitorSheet();
    var lastRow = sheet.getLastRow();
    if (lastRow < 2) return { totalVisits: 0, organic: 0, loggedIn: 0, today: 0, thisWeek: 0 };
    
    var data = sheet.getRange(2, 1, lastRow - 1, 5).getDisplayValues();
    var now = new Date();
    var today = new Date(now.getFullYear(), now.getMonth(), now.getDate());
    var weekAgo = new Date(today.getTime() - 7 * 24 * 60 * 60 * 1000);
    
    var total = data.length;
    var organic = 0;
    var loggedIn = 0;
    var todayCount = 0;
    var weekCount = 0;
    
    for (var i = 0; i < data.length; i++) {
      var timestamp = data[i][0];
      var empId = data[i][3]; // Column 4 is EmpID
      
      if (!empId || empId === 'Organic' || empId.trim() === '') organic++;
      else loggedIn++;
      
      if (timestamp) {
        var visitDate = new Date(timestamp);
        if (!isNaN(visitDate.getTime())) {
          var visitDay = new Date(visitDate.getFullYear(), visitDate.getMonth(), visitDate.getDate());
          if (visitDay.getTime() === today.getTime()) todayCount++;
          if (visitDay >= weekAgo) weekCount++;
        }
      }
    }
    
    return { totalVisits: total, organic: organic, loggedIn: loggedIn, today: todayCount, thisWeek: weekCount };
  } catch (e) { return { totalVisits: 0, organic: 0, loggedIn: 0, today: 0, thisWeek: 0 }; }
}

function getVisitorLogData() {
  try {
    var sheet = getOrCreateVisitorSheet();
    var lastRow = sheet.getLastRow();
    if (lastRow < 2) return { error: 'No visitor data' };
    
    var data = sheet.getRange(2, 1, lastRow - 1, 5).getDisplayValues();
    var rows = [];
    var dailyStats = {};
    var totalOrganic = 0;
    var totalLoggedIn = 0;
    var hours = {};
    
    for (var i = 0; i < data.length; i++) {
      var timestamp = data[i][0];
      var page = data[i][1];
      var type = data[i][2];
      var empId = data[i][3];
      var userAgent = data[i][4];
      
      var userType = (!empId || empId === 'Organic' || empId.trim() === '') ? 'Organic' : 'Logged In';
      if (userType === 'Organic') totalOrganic++;
      else totalLoggedIn++;
      
      if (timestamp) {
        var visitDate = new Date(timestamp);
        if (!isNaN(visitDate.getTime())) {
          var dateStr = visitDate.toLocaleDateString('en-GB');
          var timeStr = visitDate.toLocaleTimeString('en-GB', { hour: '2-digit', minute: '2-digit' });
          var hour = visitDate.getHours();
          
          if (!dailyStats[dateStr]) dailyStats[dateStr] = { total: 0, organic: 0, loggedIn: 0, pages: {} };
          dailyStats[dateStr].total++;
          dailyStats[dateStr][userType === 'Organic' ? 'organic' : 'loggedIn']++;
          if (!dailyStats[dateStr].pages[page]) dailyStats[dateStr].pages[page] = 0;
          dailyStats[dateStr].pages[page]++;
          
          if (!hours[hour]) hours[hour] = 0;
          hours[hour]++;
        }
      }
      
      rows.push({
        date: timestamp,
        page: page,
        type: type,
        empId: empId || 'Organic',
        userType: userType,
        userAgent: userAgent
      });
    }
    
    var dailyArray = [];
    for (var date in dailyStats) {
      dailyArray.push({
        date: date,
        total: dailyStats[date].total,
        organic: dailyStats[date].organic,
        loggedIn: dailyStats[date].loggedIn,
        pages: dailyStats[date].pages,
        hourBreakdown: hours
      });
    }
    dailyArray.sort(function(a, b) {
      var da = new Date(a.date.split('/').reverse().join('-'));
      var db = new Date(b.date.split('/').reverse().join('-'));
      return db - da;
    });
    
    return {
      visits: rows.reverse(),
      totalVisits: rows.length,
      totalOrganic: totalOrganic,
      totalLoggedIn: totalLoggedIn,
      dailyStats: dailyArray,
      hourBreakdown: hours
    };
  } catch (e) { return { error: e.toString() }; }
}

function uploadData(csvData, type, wef, remarks) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName(type);
    var configSheet = ss.getSheetByName('Config');
    var configData = configSheet.getDataRange().getValues();
    var found = false;
    for (var i = 0; i < configData.length; i++) {
      if (configData[i][0] == type) {
        configSheet.getRange(i + 1, 2, 1, 2).setValues([[wef, remarks]]);
        found = true; break;
      }
    }
    if (!found) configSheet.appendRow([type, wef, remarks]);
    sheet.clear();
    var rows = Utilities.parseCsv(csvData.replace(/\r/g, ""));
    if (rows.length > 0) {
      sheet.getRange(1, 1, rows.length, rows[0].length).setValues(rows);
      return type + " data uploaded.";
    }
  } catch (e) { return "Error: " + e.toString(); }
}

function timeToMins(timeStr) {
  if (!timeStr || typeof timeStr !== 'string' || timeStr.indexOf(':') === -1) return -1;
  var parts = timeStr.split(':');
  return (parseInt(parts[0]) || 0) * 60 + (parseInt(parts[1]) || 0);
}

function analyzeRakeRelievers(type) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName(type);
    if (!sheet || sheet.getLastRow() < 2) return {};
    
    var data = sheet.getDataRange().getDisplayValues();
    var rakeTrips = {};
    
    // Group by Rake ID (index 8)
    for (var i = 1; i < data.length; i++) {
      var rake = (data[i][8] || "").toString().trim();
      if (!rake) continue;
      if (!rakeTrips[rake]) rakeTrips[rake] = [];
      
      rakeTrips[rake].push({
        duty: data[i][0],
        depTime: data[i][10],
        arrTime: data[i][12],
        arrLoc: (data[i][11] || "").toString().trim().toUpperCase(),
        depLoc: (data[i][9] || "").toString().trim().toUpperCase(),
        rake: rake
      });
    }
    
    var gaps = {}; // Key: Rake|Time -> true
    
    // After sorting rake ID data, mark start and end of each rake's journey
    for (var rake in rakeTrips) {
      var trips = rakeTrips[rake];
      // Sort by departure time
      trips.sort(function(a, b) {
        return timeToMins(a.depTime) - timeToMins(b.depTime);
      });
      
      // Deduplicate trips (same depTime + arrTime + depLoc + arrLoc)
      var deduped = [];
      var tripSeen = {};
      for (var k = 0; k < trips.length; k++) {
        var key = trips[k].depTime + '|' + trips[k].arrTime + '|' + trips[k].depLoc + '|' + trips[k].arrLoc;
        if (!tripSeen[key]) {
          tripSeen[key] = true;
          deduped.push(trips[k]);
        }
      }
      trips = deduped;
      
      // First trip: mark departure (no person sitting in train at start)
      // Skip if depTime == arrTime (stub/placeholder entry, not a real trip)
      if (trips.length > 0) {
        if (!MathTolerant(timeToMins(trips[0].depTime), timeToMins(trips[0].arrTime))) {
          var firstDepKey = rake + '|' + trips[0].depTime;
          gaps[firstDepKey] = true;
        }
      }
      
      // Check for gaps between consecutive trips
      for (var j = 0; j < trips.length - 1; j++) {
        var currentEnd = timeToMins(trips[j].arrTime);
        var nextStart = timeToMins(trips[j+1].depTime);
        var sameDuty = trips[j].duty === trips[j+1].duty;
        
        // Exception: If current trip ends at MKPR and next trip starts from MKPR with same duty, no X needed
        var mkprException = (trips[j].arrLoc === 'MKPR' && trips[j+1].depLoc === 'MKPR' && sameDuty);
        
        if (!MathTolerant(currentEnd, nextStart)) {
          // Gap found
          if (!mkprException) {
            // Mark arrival (no reliever) and next departure (no person)
            gaps[rake + '|' + trips[j].arrTime] = true;
            gaps[rake + '|' + trips[j+1].depTime] = true;
          }
        }
      }
      
      // Last trip: mark arrival (no one sitting further in train)
      // Skip if depTime == arrTime (stub/placeholder entry, not a real trip)
      if (trips.length > 0) {
        var lastTrip = trips[trips.length - 1];
        if (!MathTolerant(timeToMins(lastTrip.depTime), timeToMins(lastTrip.arrTime))) {
          gaps[rake + '|' + lastTrip.arrTime] = true;
        }
      }
    }
    return gaps;
  } catch (e) {
    Logger.log("Reliever Analysis Error: " + e);
    return {};
  }
}

function MathTolerant(t1, t2) {
  return Math.abs(t1 - t2) <= 1;
}

function getGraphData(type, seriesArray, customDuties) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(type);
    if (!sheet || sheet.getLastRow() < 2) return { error: "No data in " + type };
    var data = sheet.getDataRange().getDisplayValues();
    var chartData = [];
    var totalMinutes = 0;
    var seenDuties = {};

    for (var i = 1; i < data.length; i++) {
      var d = (data[i][0] || "").toString().trim();
      var isSeriesMatched = seriesArray.some(function(s) { 
        if (s === '11-20') {
          var num = parseInt(d);
          return num >= 11 && num <= 20;
        }
        var num = parseInt(s);
        if (num >= 10) return d === s || d.startsWith(s + '-') || d.startsWith(s + '0');
        return d.startsWith(s);
      });
      var isCustomMatched = customDuties && customDuties.length > 0 && customDuties.indexOf(d) !== -1;
      
      if ((isSeriesMatched || isCustomMatched) && d !== "") {
        if (seenDuties[d]) continue;
        
        var timeStr = (data[i][5] || "").toString().trim();
        if (timeStr === "" || timeStr.indexOf(":") === -1) continue;
        
        var p = timeStr.split(':');
        var mins = (parseInt(p[0]) || 0) * 60 + (parseInt(p[1]) || 0);
        
        seenDuties[d] = true;
        
        chartData.push({ 
          duty: d, 
          running: mins / 60, 
          runningStr: timeStr, 
          dutyStr: (data[i][6] || "").toString(), 
          signOnLoc: (data[i][2] || "").toString(),
          signOnTime: (data[i][1] || "").toString(),
          signOffLoc: (data[i][3] || "").toString(),
          signOffTime: (data[i][4] || "").toString()
        });
        totalMinutes += mins;
      }
    }
    
    if (!chartData.length) return { error: "No duties found." };
    
    var avgMins = totalMinutes / chartData.length;
    var avgHrs = Math.floor(avgMins / 60);
    var remMins = Math.round(avgMins % 60);
    if (remMins === 60) { avgHrs++; remMins = 0; }
    var avgStr = avgHrs.toString().padStart(2, '0') + ":" + remMins.toString().padStart(2, '0');

    return { details: chartData, avgTime: avgStr, day: type };
  } catch (e) { return { error: e.toString() }; }
}

function getFormFields() {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName('FormBuilder');
    if (!sheet) return { heading: '', fields: [] };
    var data = sheet.getDataRange().getDisplayValues();
    
    var result = { heading: '', fields: [] };
    
    // Check if new format (has FormHeading in header)
    if (data.length > 1 && data[0][0] === 'FormHeading') {
      // Row 2 is the form heading
      if (data.length > 2 && data[2][0]) {
        result.heading = data[2][0];
      }
      
      // Start from row 3 for fields (skip header row 1 and heading row 2)
      for (var i = 3; i < data.length; i++) {
        if (data[i][1]) { // FieldName is in column 2
          result.fields.push({
            name: data[i][1],
            type: data[i][2] || 'text',
            required: data[i][3] === 'true',
            options: data[i][4] ? data[i][4].split(',').map(function(s){ return s.trim(); }) : []
          });
        }
      }
    } else {
      // Old format - start from row 2
      for (var i = 1; i < data.length; i++) {
        if (data[i][0]) {
          result.fields.push({
            name: data[i][0],
            type: data[i][1] || 'text',
            required: data[i][2] === 'true',
            options: data[i][3] ? data[i][3].split(',').map(function(s){ return s.trim(); }) : []
          });
        }
      }
    }
    
    return result;
  } catch (e) { return { heading: '', fields: [] }; }
}

function analyzeAndCacheTetraData(type) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var dataSheet = ss.getSheetByName(type);
    if (!dataSheet || dataSheet.getLastRow() < 2) return { success: false, message: 'No ' + type + ' data found. Upload CSV first.' };
    
    var lastRow = dataSheet.getLastRow();
    var lastCol = dataSheet.getLastColumn();
    var data = dataSheet.getDataRange().getDisplayValues();
    
    // Show debug info
    Logger.log(type + ' - Rows: ' + lastRow + ', Cols: ' + lastCol);
    if (data.length > 0) {
      Logger.log('Header row: ' + data[0].join(' | '));
    }
    
    // Count valid trips (rows with rake ID in column 9 - index 8)
    var validTrips = 0;
    var rakeList = [];
    
    // Check columns 9-13 for sample row
    if (data.length > 1) {
      Logger.log('Sample data cols 9-13: ' + [data[1][8], data[1][9], data[1][10], data[1][11], data[1][12]]);
    }
    
    for (var i = 1; i < data.length; i++) {
      var rake = (data[i][8] || "").toString().trim();
      var duty = (data[i][0] || "").toString().trim();
      if (rake && duty) {
        validTrips++;
        if (rakeList.indexOf(rake) === -1) rakeList.push(rake);
      }
    }
    
    if (validTrips === 0) return { success: false, message: type + ' - No valid data! Check column I (rake ID). Found ' + validTrips + ' trips.' };
    
    // Build full data map
    var allTrips = {};
    
    for (var i = 1; i < data.length; i++) {
      var duty = (data[i][0] || "").toString().trim();
      var rake = (data[i][8] || "").toString().trim();
      var depTime = data[i][10];
      var arrTime = data[i][12];
      var depLoc = (data[i][9] || "").toString().trim().toUpperCase();
      var arrLoc = (data[i][11] || "").toString().trim().toUpperCase();
      
      if (!rake || !duty) continue;
      
      if (!allTrips[rake]) allTrips[rake] = [];
      allTrips[rake].push({
        duty: duty,
        depTime: depTime,
        arrTime: arrTime,
        depLoc: depLoc,
        arrLoc: arrLoc
      });
    }
    
    var gaps = [];
    
    for (var rake in allTrips) {
      var trips = allTrips[rake];
      trips.sort(function(a, b) { return timeToMins(a.depTime) - timeToMins(b.depTime); });
      
      // Deduplicate trips (same depTime + arrTime + depLoc + arrLoc)
      var deduped = [];
      var tripSeen = {};
      for (var k = 0; k < trips.length; k++) {
        var key = trips[k].depTime + '|' + trips[k].arrTime + '|' + trips[k].depLoc + '|' + trips[k].arrLoc;
        if (!tripSeen[key]) {
          tripSeen[key] = true;
          deduped.push(trips[k]);
        }
      }
      trips = deduped;
      
      // First trip BOARDING — skip if depTime == arrTime (stub entry)
      if (trips.length > 0) {
        if (!MathTolerant(timeToMins(trips[0].depTime), timeToMins(trips[0].arrTime))) {
          gaps.push({ rakeId: rake, duty: trips[0].duty, station: trips[0].depLoc, time: trips[0].depTime, action: 'BOARDING' });
        }
      }
      
      for (var j = 0; j < trips.length - 1; j++) {
        var currentEnd = timeToMins(trips[j].arrTime);
        var nextStart = timeToMins(trips[j+1].depTime);
        var sameDuty = trips[j].duty === trips[j+1].duty;
        var mkprException = (trips[j].arrLoc === 'MKPR' && trips[j+1].depLoc === 'MKPR' && sameDuty);
        
        if (!MathTolerant(currentEnd, nextStart) && !mkprException) {
          gaps.push({ rakeId: rake, duty: trips[j].duty, station: trips[j].arrLoc, time: trips[j].arrTime, action: 'ALIGHTING' });
          gaps.push({ rakeId: rake, duty: trips[j+1].duty, station: trips[j+1].depLoc, time: trips[j+1].depTime, action: 'BOARDING' });
        }
      }
      
      // Last trip ALIGHTING — skip if depTime == arrTime (stub entry)
      if (trips.length > 0) {
        var lastTrip = trips[trips.length - 1];
        if (!MathTolerant(timeToMins(lastTrip.depTime), timeToMins(lastTrip.arrTime))) {
          gaps.push({ rakeId: rake, duty: lastTrip.duty, station: lastTrip.arrLoc, time: lastTrip.arrTime, action: 'ALIGHTING' });
        }
      }
    }
    
    var flatData = [];
    for (var i = 0; i < gaps.length; i++) {
      flatData.push([
        type,
        gaps[i].rakeId,
        gaps[i].duty,
        gaps[i].action === 'BOARDING' ? gaps[i].station : '-',
        gaps[i].action === 'BOARDING' ? gaps[i].time : '-',
        gaps[i].action === 'ALIGHTING' ? gaps[i].station : '-',
        gaps[i].action === 'ALIGHTING' ? gaps[i].time : '-',
        gaps[i].action
      ]);
    }
    
    var cacheSheet = getOrCreateTetraCacheSheet();
    
    // Clear all existing data for this type first
    var cacheLastRow = cacheSheet.getLastRow();
    if (cacheLastRow > 1) {
      var existingData = cacheSheet.getDataRange().getValues();
      var rowsToDelete = [];
      for (var i = 1; i < existingData.length; i++) {
        if ((existingData[i][0] || "").toString().trim() === type) {
          rowsToDelete.push(i + 1);
        }
      }
      if (rowsToDelete.length > 0) {
        rowsToDelete.sort(function(a, b) { return b - a; });
        for (var i = 0; i < rowsToDelete.length; i++) {
          cacheSheet.deleteRow(rowsToDelete[i]);
        }
      }
      // Force recalculation of last row
      cacheLastRow = cacheSheet.getLastRow();
    }
    
    // Append new data starting from row 2
    if (flatData.length > 0) {
      var startRow = cacheLastRow + 1;
      if (cacheLastRow < 2) startRow = 2;
      cacheSheet.getRange(startRow, 1, flatData.length, 8).setValues(flatData);
      Logger.log('Wrote ' + flatData.length + ' rows starting at row ' + startRow);
    }
    
    // Verify data was written
    var verifyLastRow = cacheSheet.getLastRow();
    var verifyData = cacheSheet.getDataRange().getDisplayValues();
    var writeCount = 0;
    for (var i = 1; i < verifyData.length; i++) {
      if ((verifyData[i][0] || "").toString().trim() === type) writeCount++;
    }
    Logger.log('Verification: Found ' + writeCount + ' rows for ' + type + ' (expected ' + flatData.length + ')');
    
    return { success: true, message: type + ': ' + validTrips + ' trips, ' + rakeList.length + ' rakes, ' + flatData.length + ' tetra', writeCount: writeCount, count: flatData.length };
  } catch (e) { return { success: false, message: 'Error: ' + e.toString() }; }
}

        }
      }
    }
    return { success: true, message: type + ' cache cleared' };
  } catch (e) { return { success: false, message: 'Error: ' + e.toString() }; }
}

function getLiveTetraData(type, stationFilter, directionFilter) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName(type);
    if (!sheet || sheet.getLastRow() < 2) return { error: "No data for " + type + ". Upload CSV first." };
    
    var data = sheet.getDataRange().getDisplayValues();
    var gaps = analyzeRakeRelievers(type);
    
    var seenDuties = {};
    var seenRakes = {};
    var tripMap = {};
    var depTimes = {};
    var arrTimes = {};
    
    for (var i = 1; i < data.length; i++) {
      var rake = (data[i][8] || "").toString().trim();
      var duty = (data[i][0] || "").toString().trim();
      var depTime = data[i][10];
      var arrTime = data[i][12];
      var depLoc = (data[i][9] || "").toString().trim().toUpperCase();
      var arrLoc = (data[i][11] || "").toString().trim().toUpperCase();
      
      if (!rake || !duty) continue;
      
      var key = rake + '|' + depTime;
      if (!tripMap[key]) {
        tripMap[key] = { rakeId: rake, duty: duty, boardStn: depLoc, boardTime: depTime, alightStn: arrLoc, alightTime: arrTime };
      }
      depTimes[key] = true;
      
      var arrKey = rake + '|' + arrTime;
      arrTimes[arrKey] = true;
    }
    
    var tetraData = [];
    var added = {};
    
    for (var gapKey in gaps) {
      var rakeId = gapKey.split('|')[0];
      var gapTime = gapKey.split('|')[1];
      
      var isBoard = depTimes[gapKey];
      var isAlight = arrTimes[gapKey];
      
      if (tripMap[gapKey] && !seenDuties[tripMap[gapKey].duty]) { seenDuties[tripMap[gapKey].duty] = true; }
      if (!seenRakes[rakeId]) { seenRakes[rakeId] = true; }
      
      if (isBoard && tripMap[gapKey]) {
        var item = {
          rakeId: rakeId,
          duty: tripMap[gapKey].duty,
          boardStn: tripMap[gapKey].boardStn,
          boardTime: tripMap[gapKey].boardTime,
          alightStn: tripMap[gapKey].alightStn,
          alightTime: tripMap[gapKey].alightTime,
          action: 'BOARDING'
        };
        
        var addIt = true;
        if (stationFilter && stationFilter !== 'ALL' && item.boardStn.indexOf(stationFilter) === -1) addIt = false;
        if (directionFilter && directionFilter !== 'ALL' && directionFilter !== 'BOARDING' && item.action !== directionFilter) addIt = false;
        
        if (addIt) {
          var uniqKey = rakeId + '|' + gapTime + '|BOARDING';
          if (!added[uniqKey]) {
            tetraData.push(item);
            added[uniqKey] = true;
          }
        }
      }
      
      if (isAlight) {
        var arrKey = rakeId + '|' + gapTime;
        var arrTrip = null;
        for (var k in tripMap) {
          var parts = k.split('|');
          if (parts[0] === rakeId && tripMap[k].alightTime === gapTime) {
            arrTrip = tripMap[k];
            break;
          }
        }
        
        if (arrTrip) {
          if (!seenDuties[arrTrip.duty]) { seenDuties[arrTrip.duty] = true; }
          var item = {
            rakeId: rakeId,
            duty: arrTrip.duty,
            boardStn: arrTrip.boardStn,
            boardTime: arrTrip.boardTime,
            alightStn: arrTrip.alightStn,
            alightTime: arrTrip.alightTime,
            action: 'ALIGHTING'
          };
          
          var addIt = true;
          if (stationFilter && stationFilter !== 'ALL' && item.alightStn.indexOf(stationFilter) === -1) addIt = false;
          if (directionFilter && directionFilter !== 'ALL' && directionFilter !== 'DEBOARDING' && item.action !== directionFilter && directionFilter !== 'ALIGHTING') addIt = false;
          
          if (addIt) {
            var uniqKey = rakeId + '|' + gapTime + '|ALIGHTING';
            if (!added[uniqKey]) {
              tetraData.push(item);
              added[uniqKey] = true;
            }
          }
        }
      }
    }
    
    tetraData.sort(function(a, b) {
      var timeA = timeToMins(a.action === 'BOARDING' ? a.boardTime : a.alightTime);
      var timeB = timeToMins(b.action === 'BOARDING' ? b.boardTime : b.alightTime);
      if (timeA !== timeB) return timeA - timeB;
      if (a.rakeId !== b.rakeId) return a.rakeId.localeCompare(b.rakeId);
      return 0;
    });
    
    return {
      tetraData: tetraData,
      tetraCount: tetraData.length,
      dutyCount: Object.keys(seenDuties).length,
      rakeCount: Object.keys(seenRakes).length,
      day: type,
      stationFilter: stationFilter,
      directionFilter: directionFilter,
      source: 'live'
    };
  } catch (e) { return { error: e.toString() }; }
}

function getCachedTetraData(type, stationFilter, directionFilter) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var cacheSheet = ss.getSheetByName('TetraCache');
    if (!cacheSheet || cacheSheet.getLastRow() < 2) return null;
    
    var data = cacheSheet.getDataRange().getDisplayValues();
    var tetraData = [];
    var seenDuties = {};
    var seenRakes = {};
    
    for (var i = 1; i < data.length; i++) {
      var dayType = (data[i][0] || "").toString().trim();
      if (dayType !== type) continue;
      
      var rakeId = (data[i][1] || "").toString().trim();
      var duty = (data[i][2] || "").toString().trim();
      var boardStn = (data[i][3] || "").toString().trim();
      var boardTime = (data[i][4] || "").toString();
      var alightStn = (data[i][5] || "").toString().trim();
      var alightTime = (data[i][6] || "").toString();
      var action = (data[i][7] || "").toString().trim();
      
      if (!rakeId) continue;
      
      if (duty && !seenDuties[duty]) seenDuties[duty] = true;
      if (!seenRakes[rakeId]) seenRakes[rakeId] = true;
      
      if (stationFilter && stationFilter !== 'ALL' && action === 'BOARDING' && boardStn.indexOf(stationFilter) === -1) continue;
      if (stationFilter && stationFilter !== 'ALL' && action === 'ALIGHTING' && alightStn.indexOf(stationFilter) === -1) continue;
      if (directionFilter && directionFilter !== 'ALL' && directionFilter !== 'BOARDING' && directionFilter !== 'DEBOARDING' && directionFilter !== 'ALIGHTING' && action !== directionFilter) continue;
      
      var item = {
        rakeId: rakeId,
        duty: duty,
        boardStn: boardStn,
        boardTime: boardTime,
        alightStn: alightStn,
        alightTime: alightTime,
        action: action
      };
      tetraData.push(item);
    }
    
    if (tetraData.length === 0) return null;
    
    tetraData.sort(function(a, b) {
      var timeA = timeToMins(a.action === 'BOARDING' ? a.boardTime : a.alightTime);
      var timeB = timeToMins(b.action === 'BOARDING' ? b.boardTime : b.alightTime);
      if (timeA !== timeB) return timeA - timeB;
      if (a.rakeId !== b.rakeId) return a.rakeId.localeCompare(b.rakeId);
      return 0;
    });
    
    return {
      tetraData: tetraData,
      tetraCount: tetraData.length,
      dutyCount: Object.keys(seenDuties).length,
      rakeCount: Object.keys(seenRakes).length,
      day: type,
      stationFilter: stationFilter,
      directionFilter: directionFilter,
      source: 'cache'
    };
  } catch (e) { return null; }
}

  if (cachedResult) return cachedResult;
  return getLiveTetraData(type, stationFilter, directionFilter);
}

      var existingData = cacheSheet.getDataRange().getValues();
      var rowsToDelete = [];
      for (var i = 1; i < existingData.length; i++) {
        if ((existingData[i][0] || "").toString().trim() === type) {
          rowsToDelete.push(i + 1);
        }
      }
      if (rowsToDelete.length > 0) {
        rowsToDelete.sort(function(a, b) { return b - a; });
        for (var i = 0; i < rowsToDelete.length; i++) {
          cacheSheet.deleteRow(rowsToDelete[i]);
        }
      }
      // Force recalculation
      cacheLastRow = cacheSheet.getLastRow();
    }
    
    var rows = [];
    for (var duty in dutyDataMap) {
      rows.push([type, duty, dutyDataMap[duty].signOn, dutyDataMap[duty].signOnLoc, dutyDataMap[duty].signOff, dutyDataMap[duty].signOffLoc, dutyDataMap[duty].km.toFixed(2)]);
    }
    
    // Append new data
    if (rows.length > 0) {
      var startRow = cacheLastRow + 1;
      if (cacheLastRow < 2) startRow = 2;
      cacheSheet.getRange(startRow, 1, rows.length, 7).setValues(rows);
      Logger.log('KM Wrote ' + rows.length + ' rows starting at row ' + startRow);
    }
    
    // Verify
    var verifyData = cacheSheet.getDataRange().getValues();
    var savedCount = 0;
    for (var v = 1; v < verifyData.length; v++) {
      if ((verifyData[v][0] || "").toString().trim() === type) savedCount++;
    }
    Logger.log('KM Verification: Found ' + savedCount + ' rows (expected ' + rows.length + ')');
    
    var totalKm = 0;
    for (var d in dutyDataMap) totalKm += dutyDataMap[d].km;
    
    return { success: true, message: type + ' KM data cached (' + rows.length + ' duties verified, ' + totalKm.toFixed(2) + ' km)', count: rows.length, totalKm: totalKm.toFixed(2) };
  } catch (e) { return { success: false, message: 'Error: ' + e.toString() }; }
}

function clearKMCache(type) {
  try {
    var cacheSheet = getOrCreateKMCacheSheet();
    var lastRow = cacheSheet.getLastRow();
    if (lastRow > 1) {
      var data = cacheSheet.getRange(2, 1, lastRow - 1, 1).getValues();
      var rowsToDelete = [];
      for (var i = 0; i < data.length; i++) {
        if ((data[i][0] || "").toString().trim() === type) {
          rowsToDelete.push(i + 2);
        }
      }
      rowsToDelete.sort(function(a, b) { return b - a; });
      for (var i = 0; i < rowsToDelete.length; i++) {
        cacheSheet.deleteRow(rowsToDelete[i]);
      }
    }
    return { success: true, message: type + ' KM cache cleared' };
  } catch (e) { return { success: false, message: 'Error: ' + e.toString() }; }
}

function getKMCacheStatus() {
  try {
    var cacheSheet = getOrCreateKMCacheSheet();
    var lastRow = cacheSheet.getLastRow();
    if (lastRow < 2) return { Weekday: false, Saturday: false, Sunday: false, Special: false };
    var data = cacheSheet.getRange(1, 1, lastRow, 1).getValues();
    var status = { Weekday: false, Saturday: false, Sunday: false, Special: false };
    for (var i = 1; i < data.length; i++) {
      var dayType = data[i][0];
      if (status.hasOwnProperty(dayType)) status[dayType] = true;
    }
    return status;
  } catch (e) { return { Weekday: false, Saturday: false, Sunday: false, Special: false }; }
}
