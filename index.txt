			function addQuickField() {
				var nameInput = document.getElementById('quickFieldName');
				var typeSelect = document.getElementById('quickFieldType');
				var requiredCheck = document.getElementById('quickFieldRequired');
				var optionsInput = document.getElementById('quickFieldOptions');
				
				var fieldName = nameInput.value.trim();
				if (!fieldName) return alert('Please enter a field name!');
				
				var fieldType = typeSelect.value;
				var required = requiredCheck.checked;
				var options = [];
				
				if (fieldType === 'select') {
					var opts = optionsInput.value.trim();
					if (opts) {
						options = opts.split(',').map(function(s) { return s.trim(); }).filter(function(s) { return s; });
					}
					if (options.length === 0) return alert('Please enter dropdown options (comma separated)!');
				}
				
				var field = {
					name: fieldName,
					type: fieldType,
					required: required,
					options: options
				};
				
				formFields.push(field);
				renderFormFieldsList();
				
				// Clear inputs

						}

						html += '</div>';
					});

									field.options.forEach(function(opt) {

									});
								}

							} else if (field.type === 'date') {
								previewHtml += '<input type="date" class="jarvis-input" disabled>';
							} else if (field.type === 'time') {
								previewHtml += '<input type="time" class="jarvis-input" disabled>';
							} else {
								previewHtml += '<input type="' + field.type + '" class="jarvis-input" disabled placeholder="Enter ' + field.name + '...">';
							}

						});
					}

			function showPage(id) {
				// Check authentication for protected pages
				if (id === 'pageAdmin' && !currentUser) {
					alert("⚠️ Please login first!");
					showLoginPage();
					return;
				}
				// Allow both admin and crew controller to access pageAdmin
				var userRole = currentUser && currentUser.accessLevel ? currentUser.accessLevel.toLowerCase() : '';
				if (id === 'pageAdmin' && userRole !== 'admin' && userRole !== 'crewcontroller') {
					alert("⚠️ Access denied!");
					return;
				}
				if (id === 'pageCrewController' && !currentUser) {
					alert("⚠️ Please login first!");
					showLoginPage();
					return;
				}
				
				// Track page visit
				var pageName = id.replace('page', '');
				trackVisit(pageName, 'PageView');
				
				// Show/hide header login bar based on page
				var headerBar = document.getElementById('loggedInUserHeader');
				if (headerBar) {
					if ((id === 'pageAdmin' || id === 'pageCrewController') && currentUser) {
						headerBar.classList.add('show');
					} else {
						headerBar.classList.remove('show');
					}
				}
				
				document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
				const page = document.getElementById(id);
				if (!page) return;
				page.classList.add('active');
				page.classList.remove('cinematic-enter');
				void page.offsetWidth;
				page.classList.add('cinematic-enter');
				window.scrollTo({top: 0, behavior: 'smooth'});
			}

			function formatTime(h) {
				if (isNaN(h)) return "00:00";
				let hrs = Math.floor(h);
				let mins = Math.round((h - hrs) * 60);
				if (mins === 60) { hrs++; mins = 0; }
				return hrs.toString().padStart(2, '0') + ':' + mins.toString().padStart(2, '0');
			}

					wsData.push(['Date', 'Total', 'Organic', 'Logged In']);
					
					data.dailyStats.forEach(function(day) {
						wsData.push([day.date, day.total, day.organic, day.loggedIn]);
						var pageRows = [];
						for (var p in day.pages) {
							pageRows.push(p + ': ' + day.pages[p]);
						}
						if (pageRows.length > 0) {
							wsData.push(['', 'Pages: ' + pageRows.join(', ')]);
						}
					});
					
					wsData.push(['']);
					wsData.push(['HOURLY BREAKDOWN (24-hour format)']);
					wsData.push(['Hour', 'Visits']);
					for (var h = 0; h < 24; h++) {
						var count = data.hourBreakdown[h] || 0;
						wsData.push([h + ':00 - ' + h + ':59', count]);
					}
					
					wsData.push(['']);
					wsData.push(['VISIT LOG (Latest First)']);
					wsData.push(['Date/Time', 'Page', 'Type', 'Emp ID', 'User Type', 'User Agent']);
					
					data.visits.forEach(function(v) {
						wsData.push([v.date, v.page, v.type, v.empId, v.userType, v.userAgent]);
					});
					
					var ws = XLSX.utils.aoa_to_sheet(wsData);
					var wb = XLSX.utils.book_new();
					XLSX.utils.book_append_sheet(wb, ws, 'Visitor Log');
